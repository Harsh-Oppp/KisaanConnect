⚙ Configure ESP32

Enter the local IP address of your ESP32. It must serve JSON at the /data endpoint.

▸ Sample Arduino Code for ESP32
#include <WiFi.h> #include <WebServer.h> const char* ssid = "YOUR_WIFI_SSID"; const char* password = "YOUR_WIFI_PASSWORD"; WebServer server(80); void handleData() { // Replace these values with real sensor readings float n = 140.0; float p = 45.0; float k = 120.0; float ph = 6.5; float moisture = 65.0; float temperature = 28.0; String json = "{"; json += "\"n\":" + String(n,1) + ","; json += "\"p\":" + String(p,1) + ","; json += "\"k\":" + String(k,1) + ","; json += "\"ph\":" + String(ph,2) + ","; json += "\"moisture\":" + String(moisture,1) + ","; json += "\"temperature\":" + String(temperature,1); json += "}"; server.sendHeader("Access-Control-Allow-Origin", "*"); server.send(200, "application/json", json); } void setup() { Serial.begin(115200); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nWiFi connected. IP: " + WiFi.localIP().toString()); server.on("/data", handleData); server.begin(); } void loop() { server.handleClient(); }
